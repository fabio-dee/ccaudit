**2. `apps/ccaudit/src/cli/index.ts`** -- gunshi CLI router:
```typescript
import { cli } from 'gunshi';
import { ghostCommand } from './commands/ghost.ts';

export async function run(): Promise<void> {
  let args = process.argv.slice(2);
  // Handle npx double-name edge case: `npx ccaudit ccaudit ghost` -> strip first `ccaudit`
  if (args[0] === 'ccaudit') args = args.slice(1);

  await cli(args, ghostCommand, {
    name: 'ccaudit',
    version: '0.0.1',
    description: 'Audit Claude Code ghost inventory — agents, skills, MCP servers, and memory files',
    subCommands: {
      ghost: ghostCommand,
      // Future: inventory, mcp, trend, restore, contribute
    },
  });
}
```

NOTE: Using hardcoded version `'0.0.1'` instead of JSON import for Phase 1 simplicity. Phase 6 will switch to reading from package.json when polishing for release.

**1. `.github/workflows/ci.yaml`**:
```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm
      - run: pnpm install --frozen-lockfile
      - run: pnpm lint
      - run: pnpm format:check

  typecheck:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm
      - run: pnpm install --frozen-lockfile
      - run: pnpm typecheck

  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm
      - run: pnpm install --frozen-lockfile
      - run: pnpm test

  build:
    runs-on: ubuntu-latest
    needs: [lint, typecheck, test]
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: pnpm
      - run: pnpm install --frozen-lockfile
      - run: pnpm -r build
      # Verify binary executes
      - run: node apps/ccaudit/dist/index.js --help
      # Verify shebang preserved
      - run: head -1 apps/ccaudit/dist/index.js | grep -q '#!/usr/bin/env node'
      # Verify in-source tests stripped
      - run: "! grep -q 'import.meta.vitest' apps/ccaudit/dist/index.js"
      # Verify zero runtime deps in pack output
      - run: cd apps/ccaudit && npm pack --dry-run 2>&1
```

The `build` job runs after lint, typecheck, and test pass (`needs: [lint, typecheck, test]`). It verifies all Phase 1 success criteria:
- Binary executes (`node dist/index.js --help`)
- Shebang preserved in output
- In-source tests stripped from production
- Zero runtime deps via `npm pack --dry-run`
