    **1. Create `packages/internal/src/token/file-size-estimator.ts`:**

    ```typescript
    import { stat } from 'node:fs/promises';
    import type { TokenEstimate } from './types.ts';

    export const BYTES_PER_TOKEN = 4;

    export async function estimateFromFileSize(filePath: string): Promise<TokenEstimate | null> {
      try {
        const s = await stat(filePath);
        const tokens = Math.ceil(s.size / BYTES_PER_TOKEN);
        return {
          tokens,
          confidence: 'estimated',
          source: `file size (${s.size} bytes / ${BYTES_PER_TOKEN} bytes per token)`,
        };
      } catch {
        return null;
      }
    }
    ```

    Add in-source vitest tests. Create a temp file with known content (e.g., 400 bytes = "a".repeat(400)), verify tokens = 100. Test nonexistent path returns null.

    **2. Create `packages/internal/src/token/format.ts`:**

    Import `TokenEstimate` from `./types.ts`.

    `formatTokenEstimate(estimate: TokenEstimate | null): string`:
    - null -> "unknown"
    - tokens >= 10000 -> `~${Math.round(tokens / 1000)}k tokens (${confidence})`
    - tokens >= 1000 -> `~${(tokens / 1000).toFixed(1)}k tokens (${confidence})`
    - tokens < 1000 -> `~${tokens} tokens (${confidence})`

    `formatTotalOverhead(totalTokens: number, contextWindowSize: number = 200_000): string`:

    - formatted: tokens >= 1000 -> `~${Math.round(tokens / 1000)}k`, else `~${tokens}`
    - return `${formatted} tokens (~${percentage}% of ${contextWindowSize / 1000}k context window)`
