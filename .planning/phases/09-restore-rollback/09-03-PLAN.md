    3. **gunshi command definition:**
    ```typescript
    export const restoreCommand = define({
      name: 'restore',
      description: 'Restore items archived by a previous --dangerously-bust-ghosts run',
      toKebab: true,
      renderHeader: null, // Phase 7 precedent: suppress decorative banner for structured output modes
      args: {
        ...outputArgs,
        list: {
          type: 'boolean',
          description: 'List all archived items across all busts (read-only)',
          default: false,
        },
      },
      async run(ctx) {
        initColor();
        const outMode = resolveOutputMode(ctx.values);

        // Parse invocation: positional name vs --list vs full restore
        const positionalName = (ctx._ ?? [])[0] ?? null;
        const listFlag = ctx.values.list === true;

        const mode: RestoreMode = listFlag
          ? { kind: 'list' }
          : positionalName !== null
            ? { kind: 'single', name: positionalName }
            : { kind: 'full' };

        const warnings: string[] = [];
        const deps = buildProductionRestoreDeps(warnings);

        let result: RestoreResult;
        try {
          result = await executeRestore(mode, deps);
        } catch (err) {
          // Fatal error path — shouldn't happen with injectable deps but handle defensively
          const message = err instanceof Error ? err.message : String(err);
          if (outMode.json) {
            process.stdout.write(
              JSON.stringify(buildJsonEnvelope('restore', 'n/a', 2, { error: message })) + '\n',
            );
          } else {
            process.stderr.write(`ccaudit restore failed: ${message}\n`);
          }
          process.exit(2);
        }

        const exitCode = restoreResultToExitCode(result);

        // Output mode matrix
        if (outMode.json) {
          process.stdout.write(
            JSON.stringify(buildJsonEnvelope('restore', 'n/a', exitCode, restoreResultToJson(result, warnings))) + '\n',
          );
        } else if (outMode.csv) {
          process.stdout.write(renderRestoreCsv(result, warnings));
        } else if (outMode.quiet) {
          process.stdout.write(renderRestoreQuiet(result));
        } else {
          process.stdout.write(renderRestoreRendered(result, warnings, outMode.verbose ?? false));
        }

        process.exit(exitCode);
      },
    });
    ```

    function restoreResultToJson(result: RestoreResult, warnings: string[]): object {
      const base = { warnings };
      switch (result.status) {
        case 'success':
        case 'partial-success':
          return {
            ...base,
            status: result.status,
            counts: result.counts,
            manifest_path: result.manifestPath,
            duration_ms: result.duration_ms,
            failed: result.status === 'partial-success' ? result.failed : 0,
          };
        case 'no-manifests':
          return { ...base, status: 'no-manifests', message: 'No bust history found. Run ccaudit --dangerously-bust-ghosts first.' };
        case 'name-not-found':
          return { ...base, status: 'name-not-found', name: result.name, message: `No archived item named '${result.name}' found.` };
        case 'manifest-corrupt':
          return { ...base, status: 'manifest-corrupt', path: result.path };
        case 'list':
          return { ...base, status: 'list', entries: result.entries.map(summarizeListEntry) };
        case 'running-process':
          return { ...base, status: 'running-process', pids: result.pids, self_invocation: result.selfInvocation, message: result.message };
        case 'process-detection-failed':
          return { ...base, status: 'process-detection-failed', error: result.error };
        case 'config-parse-error':
        case 'config-write-error':
          return { ...base, status: result.status, path: result.path, error: result.error };
      }
    }

    function summarizeListEntry(entry: ManifestListEntry): object {
      return {
        path: entry.path,
        mtime: entry.mtime.toISOString(),
        is_partial: entry.isPartial,
        op_count: entry.opCount,
        items: entry.ops.filter((o): o is ArchiveOp | DisableOp => o.op_type === 'archive' || o.op_type === 'disable').map(summarizeOp),
      };
    }

    function summarizeOp(op: ArchiveOp | DisableOp): object {
      if (op.op_type === 'archive') {
        return {
          category: op.category,
          name: path.basename(op.archive_path, path.extname(op.archive_path)),
          source_path: op.source_path,
          archive_path: op.archive_path,
        };
      }
      return {
        category: 'mcp',
        name: extractServerName(op.original_key),
        config_path: op.config_path,
        disabled_key: op.new_key,
      };
    }

    6. **renderRestoreRendered()** — default human output per D-16:
    ```typescript
    function renderRestoreRendered(result: RestoreResult, warnings: string[], verbose: boolean): string {
      const lines: string[] = [];

      // Special modes first
      if (result.status === 'no-manifests') {
        return 'No bust history found. Run ccaudit --dangerously-bust-ghosts first.\n';
      }
      if (result.status === 'name-not-found') {
        return `No archived item named '${result.name}' found.\n`;
      }
      if (result.status === 'list') {
        return renderListOutput(result.entries);
      }
      if (result.status === 'running-process') {
        return result.message + '\n';
      }
      if (result.status === 'process-detection-failed') {
        return `Process detection failed: ${result.error}\n`;
      }
      if (result.status === 'manifest-corrupt') {
        return `Manifest is corrupt (no header record). Cannot restore from ${path.basename(result.path)}.\n`;
      }
      if (result.status === 'config-parse-error') {
        return `Failed to parse ${result.path}: ${result.error}\n`;
      }
      if (result.status === 'config-write-error') {
        return `Failed to write ${result.path}: ${result.error}\n`;
      }

      // success / partial-success
      const header = renderHeader(`Restore — ${new Date().toISOString()}`);
      lines.push(header);

      const c = result.counts;
      lines.push(`${c.unarchived.completed} agents/skills unarchived${c.unarchived.failed > 0 ? ` (${c.unarchived.failed} failed)` : ''}`);
      lines.push(`${c.reenabled.completed} MCP servers re-enabled${c.reenabled.failed > 0 ? ` (${c.reenabled.failed} failed)` : ''}`);
      lines.push(`${c.stripped.completed} memory files stripped${c.stripped.failed > 0 ? ` (${c.stripped.failed} failed)` : ''}`);

      if (verbose && warnings.length > 0) {
        lines.push('');
        lines.push('Warnings:');
        for (const w of warnings) lines.push(`  ${w}`);
      }

      lines.push('');
      if (result.status === 'success') {
        lines.push(colorize('✓ All operations completed.', 'green'));
      } else {
        lines.push(colorize(`⚠ Partial success: ${result.failed} operation(s) failed.`, 'yellow'));
      }

      return lines.join('\n') + '\n';
    }
    ```

    7. **renderListOutput()** — D-04 grouped bust listing:
    ```typescript
    function renderListOutput(entries: ManifestListEntry[]): string {
      if (entries.length === 0) {
        return 'No bust history found.\n';
      }
      const lines: string[] = [];
      lines.push(`Archived items — ${entries.length} bust${entries.length === 1 ? '' : 's'}`);
      lines.push('');
      for (const entry of entries) {
        const isoName = path.basename(entry.path, '.jsonl').replace(/^bust-/, '');
        const statusLabel = entry.isPartial ? 'partial bust' : 'clean bust';
        lines.push(`● ${isoName}  (${statusLabel}, ${entry.opCount} items)`);
        for (const op of entry.ops) {
          if (op.op_type === 'archive') {
            const name = path.basename(op.archive_path, path.extname(op.archive_path));
            lines.push(`  ${op.category.padEnd(8)}${name.padEnd(28)}${op.archive_path}`);
          } else if (op.op_type === 'disable') {
            const server = extractServerName(op.original_key);
            lines.push(`  mcp     ${server.padEnd(28)}${op.config_path} (key: ${op.new_key})`);
          } else if (op.op_type === 'flag' || op.op_type === 'refresh') {
            lines.push(`  memory  ${path.basename(op.file_path).padEnd(28)}${op.file_path} (frontmatter)`);
          }
          // skipped ops: omit from list per RESEARCH Q3
        }
        lines.push('');
      }
      return lines.join('\n');
    }
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-14 | Tampering | Positional name argument injection (e.g. `restore ../../../etc/passwd`) | mitigate | Plan 01's findManifestForName uses path.basename of archive_path for matching, so even a path-like `<name>` gets basenamed before comparison. The name never reaches fs APIs directly — it's only used as a lookup key. Verified by Plan 04 integration test. |
| T-09-15 | Information Disclosure | --json envelope leaks absolute paths | accept | Paths in JSON output are the user's own file system; same privacy model as ghost --json. No remote endpoint involved. |
| T-09-16 | Denial of Service | Very large manifest → JSON envelope balloons memory | mitigate | --list mode emits summarized ops (no full JSONL inlined); per-op details limited to category + name + path. Full manifest body is only read via readManifest (streaming line-based reader from Phase 8). |
| T-09-17 | Elevation of Privilege | Wrong exit code allows CI pipeline to proceed past a failure | mitigate | Exhaustive switch in restoreResultToExitCode with no default case — TypeScript will fail compilation if RestoreResult gains a new variant without updating the switch. | T-09-18 | Spoofing | --list output consumed by automation could be spoofed by TTY-specific control sequences | mitigate | --list rendering does not emit escape sequences when initColor detects --no-color or NO_COLOR env; picocolors identity functions from Phase 6 gap #4 prevent ANSI injection.
