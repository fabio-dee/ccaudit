    ```typescript
    async function runRestore(
      tmpHome: string,
      flags: string[],
      opts: { timeout?: number } = {},
    ): Promise<RunResult> {
      return new Promise((resolve, reject) => {
        const originalPath = process.env.PATH ?? '';
        const child = spawn('node', [distPath, 'restore', ...flags], {
          env: {
            ...process.env,
            HOME: tmpHome,
            USERPROFILE: tmpHome,
            XDG_CONFIG_HOME: path.join(tmpHome, '.config'),
            NO_COLOR: '1',
            PATH: `${path.join(tmpHome, 'bin')}:${originalPath}`,
          },
          stdio: ['pipe', 'pipe', 'pipe'],
        });
        child.stdin.end(); // isTTY=false
        let stdout = '';
        let stderr = '';
        child.stdout.on('data', (d) => (stdout += d.toString()));
        child.stderr.on('data', (d) => (stderr += d.toString()));
        const timer = setTimeout(() => {
          child.kill('SIGKILL');
          reject(new Error(`restore command timeout after ${opts.timeout ?? 30_000}ms`));
        }, opts.timeout ?? 30_000);
        child.on('close', (code) => {
          clearTimeout(timer);
          resolve({ stdout, stderr, exitCode: code });
        });
        child.on('error', (err) => {
          clearTimeout(timer);
          reject(err);
        });
      });
    }
    ```

    ```typescript
  PID COMMAND
${lines}
EOF
`;

      // Also provide tasklist stub for Windows test environments (no-op on macOS/Linux)
    ```

    ```typescript
    async function buildBasicFixture(options: {
      includeArchive?: boolean;
      includeFooter?: boolean;
      includeHeader?: boolean;
      tamperArchive?: boolean;
    } = {}): Promise<FixtureResult> {
      const tmpHome = await setupEmptyHome();
      const archivedDir = path.join(tmpHome, '.claude', 'agents', '_archived');
      await mkdir(archivedDir, { recursive: true });

      const archivedAgentPath = path.join(archivedDir, 'code-reviewer.md');
      const agentSourcePath = path.join(tmpHome, '.claude', 'agents', 'code-reviewer.md');
      const content = '# Code Reviewer\n\nA sample agent.\n';
      if (options.includeArchive !== false) {
        const writeContent = options.tamperArchive === true ? content + '\nTAMPERED' : content;
        await writeFile(archivedAgentPath, writeContent);
      }
      const contentSha256 = createHash('sha256').update(content).digest('hex');

      // Write manifest JSONL directly (simulating a completed Phase 8 bust)
      const manifestPath = path.join(
        tmpHome, '.claude', 'ccaudit', 'manifests',
        'bust-2026-04-05T18-30-00Z.jsonl',
      );
      const lines: string[] = [];
      if (options.includeHeader !== false) {
        lines.push(JSON.stringify({
          record_type: 'header',
          started_at: '2026-04-05T18:30:00.000Z',
          ccaudit_version: '1.2.0',
          node_version: process.version,
          since: '30d',
          os: process.platform,
          checkpoint_hash: 'fake-hash',
        }));
      }
      if (options.includeArchive !== false) {
        lines.push(JSON.stringify({
          op_type: 'archive',
          category: 'agent',
          source_path: agentSourcePath,
          archive_path: archivedAgentPath,
          content_sha256: contentSha256,
          ghost_tier: 'definite-ghost',
          last_used: null,
          token_estimate: 500,
        }));
      }
      if (options.includeFooter !== false) {
        lines.push(JSON.stringify({
          record_type: 'footer',
          completed_at: '2026-04-05T18:30:10.000Z',
          counts: { archived: 1, disabled: 0, flagged: 0, skipped: 0 },
          status: 'success',
        }));
      }
      await writeFile(manifestPath, lines.join('\n') + (lines.length > 0 ? '\n' : ''));

      return { tmpHome, manifestPath, archivedAgentPath, agentSourcePath, contentSha256 };
    }

    async function teardown(tmpHome: string): Promise<void> {
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Test fixture files → dist binary under test | Tests control manifest content; no attacker input |
| subprocess stdout → test assertions | Parseable structured output required for test reliability |
| tmpHome isolation → production ~/.claude | HOME env override ensures tests cannot pollute real user data |
| Fake ps binary → restore process gate | Dev convenience; production uses real ps/tasklist |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-20 | Tampering | Test fixture manifest poisoning during CI | accept | Test fixtures are checked into source control via test file. Any poisoning visible in git diff. Same model as Phase 8 bust tests. |
| T-09-21 | Information Disclosure | Test spawns process with process.env spread — leaks user env | mitigate | Follow bust-command.test.ts precedent: spread process.env but override HOME/PATH/NO_COLOR. Tests cannot exfiltrate env because they run in the developer's own context. |
| T-09-22 | Denial of Service | Zombie subprocess after timeout | mitigate | 30s timeout with SIGKILL cleanup (matches bust-command.test.ts). Timer cleared on close event. |
| T-09-23 | Spoofing | Fake ps script allows test to fake process list | accept | Test-only feature by design. Real production uses defaultProcessDeps with real spawn('ps', ...). PATH override is the test vector. Not a production vulnerability. |
| T-09-24 | Elevation of Privilege | Test runs as user; could affect real files if HOME override fails | mitigate | Every test asserts `tmpHome` is set before operations; afterEach unconditional teardown. mkdtemp ensures collision-free paths. |
| T-09-25 | Repudiation | Test failure logs don't capture enough context for post-mortem | mitigate | Tests capture stdout + stderr separately and assert on both when relevant; vitest exit 1 shows full output on failure. Sufficient for CI debug. |
