    export async function atomicWriteJson<T>(
      targetPath: string,
      value: T,
      options: AtomicWriteOptions = {},
    ): Promise<void> {
      const opts = { ...DEFAULTS, ...options };
      const dir = path.dirname(targetPath);
      // Random 8 char suffix (not crypto-random; just collision-avoidance for concurrent callers)
      const rand = Math.random().toString(36).slice(2, 10).padEnd(8, '0');
      const tmpPath = `${targetPath}.tmp-${process.pid}-${rand}`;

      // Create parent with restrictive mode (POSIX only; no-op on Windows)
      await mkdir(dir, { recursive: true, mode: opts.dirMode });
      const body = JSON.stringify(value, null, 2);

      try {
        await writeFile(tmpPath, body, { mode: opts.mode, encoding: 'utf8' });
        await renameWithRetry(tmpPath, targetPath, opts);
      } catch (err) {
        // Best-effort cleanup of the tmp file (swallow unlink errors)
        try { await unlink(tmpPath); } catch { /* swallow */ }
        throw err;
      }
    }

    **`renameWithRetry` implementation (graceful-fs canonical pattern, adapted for CLI):**
    ```typescript
    export async function renameWithRetry(
      from: string,
      to: string,
      options: Pick<AtomicWriteOptions, 'retryTotalMs' | 'retryInitialMs' | 'retryMaxMs'> = {},
    ): Promise<void> {
      const opts = {
        retryTotalMs: options.retryTotalMs ?? DEFAULTS.retryTotalMs,
        retryInitialMs: options.retryInitialMs ?? DEFAULTS.retryInitialMs,
        retryMaxMs: options.retryMaxMs ?? DEFAULTS.retryMaxMs,
      };
      const start = Date.now();
      let backoff = opts.retryInitialMs;
      while (true) {
        try {
          await rename(from, to);
          return;
        } catch (err) {
          const code = (err as NodeJS.ErrnoException).code;
          const retryable = code === 'EPERM' || code === 'EACCES' || code === 'EBUSY';
          const elapsed = Date.now() - start;
          // Only retry on Windows + retryable code + still within budget
          if (!retryable || elapsed >= opts.retryTotalMs || process.platform !== 'win32') {
            throw err;
          }
          // graceful-fs stat-before-retry: if destination exists, original error is real
          try {
            await stat(to);
            // Destination exists — the rename actually failed for a real reason
            throw err;
          } catch (statErr) {
            if ((statErr as NodeJS.ErrnoException).code !== 'ENOENT') {
              // stat threw for a different reason (EACCES etc) — propagate original
              throw err;
            }
            // ENOENT confirms dest does not exist — safe to retry after backoff
          }
          await new Promise((r) => setTimeout(r, backoff));
          backoff = Math.min(backoff + 10, opts.retryMaxMs);
        }
      }
    }
    ```
