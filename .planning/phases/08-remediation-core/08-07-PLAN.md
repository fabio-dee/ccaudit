    /**
     * Subprocess integration tests for `ccaudit --dangerously-bust-ghosts`.
     *
     * Spawns the built binary (apps/ccaudit/dist/index.js) with HOME overridden
     * to a tmpdir fixture, asserts exit codes, stdout/stderr, manifest contents,
     * and on-disk side effects. Mirrors the pattern from dry-run-command.test.ts.
     *
     * Coverage:
     *   RMED-01: flag registered and routed through ghost command
     *   RMED-10 (D-15, D-16, D-17): confirmation ceremony, --yes-proceed-busting, non-TTY
     *   Exit code ladder: 0 (success/aborted), 1 (checkpoint/hash), 3 (running-process), 4 (non-TTY)
     *   Output matrix: --json honored, --csv rejected, --quiet suppresses progress, --ci implies --yes
     */
    import { describe, it, expect, beforeAll, beforeEach, afterEach } from 'vitest';
    import { spawn } from 'node:child_process';
    import { mkdtemp, mkdir, writeFile, rm, readFile } from 'node:fs/promises';
    import { existsSync } from 'node:fs';
    import { tmpdir } from 'node:os';
    import path from 'node:path';
    import { fileURLToPath } from 'node:url';

    const here = path.dirname(fileURLToPath(import.meta.url));
    const distPath = path.resolve(here, '..', '..', 'dist', 'index.js');

    interface RunResult {
      code: number | null;
      stdout: string;
      stderr: string;
    }

    async function runBustCommand(
      tmpHome: string,
      flags: string[],
      opts: { stdin?: string | null; timeout?: number } = {},
    ): Promise<RunResult> {
      return new Promise((resolve, reject) => {
        const child = spawn('node', [distPath, ...flags], {
          env: {
            ...process.env,
            HOME: tmpHome,
            USERPROFILE: tmpHome,
            NO_COLOR: '1',
            XDG_CONFIG_HOME: path.join(tmpHome, '.config'),
          },
          // When opts.stdin is a string we pipe stdin; when null we inherit (null = piped empty)
          stdio: [opts.stdin === undefined ? 'ignore' : 'pipe', 'pipe', 'pipe'],
        });

        let stdout = '';
        let stderr = '';
        let killed = false;
        const timer = setTimeout(() => {
          killed = true;
          child.kill('SIGKILL');
          reject(new Error(`runBustCommand timed out after ${opts.timeout ?? 30000}ms`));
        }, opts.timeout ?? 30000);

        child.stdout.on('data', (c) => { stdout += c.toString(); });
        child.stderr.on('data', (c) => { stderr += c.toString(); });
        child.on('error', (err) => { clearTimeout(timer); reject(err); });
        child.on('close', (code) => {
          clearTimeout(timer);
          if (!killed) resolve({ code, stdout, stderr });
        });

        if (opts.stdin !== undefined && opts.stdin !== null) {
          child.stdin?.write(opts.stdin);
          child.stdin?.end();
        } else if (opts.stdin === null) {
          child.stdin?.end();
        }
      });
    }

    async function buildEmptyFixture(tmpHome: string): Promise<void> {
      // Create the directories the scanner walks so it doesn't error
      await mkdir(path.join(tmpHome, '.claude'), { recursive: true });
      await mkdir(path.join(tmpHome, '.claude', 'agents'), { recursive: true });
      await mkdir(path.join(tmpHome, '.claude', 'skills'), { recursive: true });
      await mkdir(path.join(tmpHome, '.config', 'claude'), { recursive: true });
      // Empty ~/.claude.json so readClaudeConfig returns {}
      await writeFile(path.join(tmpHome, '.claude.json'), '{}', 'utf8');
    }

    async function runDryRunFirst(tmpHome: string): Promise<RunResult> {
      // Create the checkpoint by running --dry-run with --yes-proceed-busting N/A
      return runBustCommand(tmpHome, ['--dry-run', '--json'], { stdin: null });
    }

    describe('ccaudit --dangerously-bust-ghosts (integration)', () => {
      let tmpHome: string;

      beforeAll(() => {
        if (!existsSync(distPath)) {
          throw new Error(
            `dist binary not found at ${distPath}. Run \`pnpm -F ccaudit build\` before running this test.`,
          );
        }
      });

      beforeEach(async () => {
        tmpHome = await mkdtemp(path.join(tmpdir(), 'bust-cmd-'));
      });

      afterEach(async () => {
        await rm(tmpHome, { recursive: true, force: true });
      });

      // ── Exit code 4: non-TTY without --yes-proceed-busting ──
      describe('exit code 4: non-TTY requires --yes-proceed-busting (D-17)', () => {
        it('piped stdin (empty) without --yes-proceed-busting → exit 4', async () => {
          await buildEmptyFixture(tmpHome);
          // Run dry-run first so checkpoint exists (so we pass gates 1-2)
          await runDryRunFirst(tmpHome);
          const result = await runBustCommand(tmpHome, ['--dangerously-bust-ghosts'], {
            stdin: '', // piped empty stdin — isTTY false
          });
          expect(result.code).toBe(4);
          expect(result.stderr).toMatch(/requires an interactive terminal/);
          expect(result.stderr).toMatch(/--yes-proceed-busting/);
        });

        it('piped stdin WITH --yes-proceed-busting → bypasses prompt (no exit 4)', async () => {
          await buildEmptyFixture(tmpHome);
          await runDryRunFirst(tmpHome);
          const result = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--yes-proceed-busting'],
            { stdin: '' },
          );
          // Empty fixture → no ops to run → exit 0 (success)
          expect(result.code).toBe(0);
        });

        it('--ci --dangerously-bust-ghosts implies --yes-proceed-busting', async () => {
          await buildEmptyFixture(tmpHome);
          await runDryRunFirst(tmpHome);
          const result = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--ci'],
            { stdin: '' },
          );
          expect(result.code).toBe(0);
          // --ci implies --json, so stdout should be parseable JSON with bust envelope
          const parsed = JSON.parse(result.stdout);
          expect(parsed).toHaveProperty('meta');
          expect(parsed).toHaveProperty('bust');
          expect((parsed.bust as Record<string, unknown>).status).toBe('success');
        });
      });

      // ── Exit code 1: checkpoint gate failures ──
      describe('exit code 1: checkpoint and hash gate failures (D-01)', () => {
        it('no checkpoint → exit 1 with checkpoint-missing message', async () => {
          await buildEmptyFixture(tmpHome);
          // Do NOT run --dry-run first, so no checkpoint file exists
          const result = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--yes-proceed-busting'],
            { stdin: '' },
          );
          expect(result.code).toBe(1);
          expect(result.stderr).toMatch(/No checkpoint found/);
          expect(result.stderr).toMatch(/ccaudit --dry-run/);
        });

        it('hash mismatch (inventory changed) → exit 1', async () => {
          await buildEmptyFixture(tmpHome);
          await runDryRunFirst(tmpHome);
          // Now add an agent file to change the inventory after dry-run was written
          await writeFile(
            path.join(tmpHome, '.claude', 'agents', 'new-ghost.md'),
            '# new agent',
            'utf8',
          );
          const result = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--yes-proceed-busting'],
            { stdin: '' },
          );
          // New agent is brand-new → scanner marks it as definite-ghost → hash changes
          // OR new agent is recent → lastUsed = never → still likely/definite ghost
          // Either way hash differs from checkpoint
          expect(result.code).toBe(1);
          // stderr OR stdout mentions hash mismatch
          const output = result.stderr + result.stdout;
          expect(output).toMatch(/hash.*changed|inventory.*changed|hash-mismatch/i);
        });

        it('--csv on bust → exit 1 with rejection message', async () => {
          await buildEmptyFixture(tmpHome);
          await runDryRunFirst(tmpHome);
          const result = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--yes-proceed-busting', '--csv'],
            { stdin: '' },
          );
          expect(result.code).toBe(1);
          expect(result.stderr).toMatch(/--csv is not supported/);
        });
      });

      // ── Exit code 0: success paths ──
      describe('exit code 0: successful bust paths', () => {
        it('empty fixture + --yes-proceed-busting → exit 0 with manifest', async () => {
          await buildEmptyFixture(tmpHome);
          await runDryRunFirst(tmpHome);
          const result = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--yes-proceed-busting', '--json'],
            { stdin: '' },
          );
          expect(result.code).toBe(0);
          const parsed = JSON.parse(result.stdout);
          expect(parsed.meta).toBeTruthy();
          expect(parsed.bust).toBeTruthy();
          expect((parsed.bust as { status: string }).status).toBe('success');
          // Manifest file must exist
          const manifestPath = (parsed.bust as { manifestPath: string }).manifestPath;
          expect(existsSync(manifestPath)).toBe(true);
          // Manifest contains header + footer
          const manifestContent = await readFile(manifestPath, 'utf8');
          const lines = manifestContent.split('\n').filter(Boolean);
          expect(lines.length).toBeGreaterThanOrEqual(2); // header + footer
          const firstLine = JSON.parse(lines[0]!);
          expect(firstLine.record_type).toBe('header');
          const lastLine = JSON.parse(lines[lines.length - 1]!);
          expect(lastLine.record_type).toBe('footer');
          expect(lastLine.status).toBe('completed');
        });

        it('--quiet suppresses progress output, still exits 0', async () => {
          await buildEmptyFixture(tmpHome);
          await runDryRunFirst(tmpHome);
          const result = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--yes-proceed-busting', '--quiet'],
            { stdin: '' },
          );
          expect(result.code).toBe(0);
          // stdout should be empty or near-empty under --quiet (no "Done." line)
          expect(result.stdout).not.toMatch(/Done\./);
        });
      });

      // ── Exit code 3: running-process detection ──
      describe('exit code 3: running-process preflight (D-03)', () => {
        // Note: we cannot reliably spawn a "claude" process on CI runners.
        // This test asserts the exit-code PATH exists by relying on the fact that
        // process-detection-failed (spawn failure) also maps to exit 3, and we
        // can trigger spawn-failure by making `ps`/`tasklist` unavailable via PATH.
        //
        // Skipped on Windows because PATH manipulation to hide tasklist.exe is
        // unreliable in GitHub Actions Windows runners.
        it.skipIf(process.platform === 'win32')(
          'empty PATH so ps is unreachable → exit 3 (process-detection-failed)',
          async () => {
            await buildEmptyFixture(tmpHome);
            await runDryRunFirst(tmpHome);
            // Spawn with PATH stripped
            const child = spawn('node', [distPath, '--dangerously-bust-ghosts', '--yes-proceed-busting'], {
              env: {
                HOME: tmpHome,
                USERPROFILE: tmpHome,
                NO_COLOR: '1',
                XDG_CONFIG_HOME: path.join(tmpHome, '.config'),
                PATH: '/nonexistent-dir-only',
              },
              stdio: ['pipe', 'pipe', 'pipe'],
            });
            let stderr = '';
            child.stderr.on('data', (c) => { stderr += c.toString(); });
            child.stdin.end();
            const code: number | null = await new Promise((resolve) => {
              child.on('close', (c) => resolve(c));
            });
            expect(code).toBe(3);
            expect(stderr).toMatch(/Could not verify Claude Code is stopped|process-detection-failed/);
          },
        );
      });

      // ── Full pipeline with real ghost items ──
      describe('full pipeline: archive + disable + flag', () => {
        it('real fixture: 1 agent + 1 MCP + memory → manifest has 3+ ops, side effects on disk', async () => {
          // Build a fixture with a ghost agent, ghost MCP, and an old memory file
          await mkdir(path.join(tmpHome, '.claude', 'agents'), { recursive: true });
          await writeFile(
            path.join(tmpHome, '.claude', 'agents', 'ghost-agent.md'),
            '# ghost agent body',
            'utf8',
          );
          await writeFile(
            path.join(tmpHome, '.claude.json'),
            JSON.stringify({
              mcpServers: {
                'ghost-mcp': { command: 'npx', args: ['ghost'] },
              },
            }),
            'utf8',
          );
          // Old memory file (set mtime to 40 days ago so scanner classifies as stale)
          const memoryPath = path.join(tmpHome, '.claude', 'CLAUDE.md');
          await writeFile(memoryPath, '# old memory\n', 'utf8');
          // Backdate mtime by 40 days
          const oldTime = new Date(Date.now() - 40 * 24 * 60 * 60 * 1000);
          const { utimes } = await import('node:fs/promises');
          await utimes(memoryPath, oldTime, oldTime);

          // Create the XDG config dir so dual-path scanner doesn't get confused
          await mkdir(path.join(tmpHome, '.config', 'claude'), { recursive: true });

          // Run dry-run to produce checkpoint
          const dryRun = await runDryRunFirst(tmpHome);
          expect([0, 1]).toContain(dryRun.code); // 0 if no ghosts, 1 if ghosts found

          // Run bust
          const bust = await runBustCommand(
            tmpHome,
            ['--dangerously-bust-ghosts', '--yes-proceed-busting', '--json'],
            { stdin: '' },
          );
          expect([0, 1]).toContain(bust.code); // success or partial-success

          // Assert ghost-agent was moved to _archived/
          const archivedPath = path.join(tmpHome, '.claude', 'agents', '_archived', 'ghost-agent.md');
          expect(existsSync(archivedPath)).toBe(true);
          expect(existsSync(path.join(tmpHome, '.claude', 'agents', 'ghost-agent.md'))).toBe(false);

          // Assert ~/.claude.json has key-rename
          const updatedConfig = JSON.parse(await readFile(path.join(tmpHome, '.claude.json'), 'utf8'));
          expect(updatedConfig.mcpServers?.['ghost-mcp']).toBeUndefined();
          // The disabled key is ccaudit-disabled:ghost-mcp (first-time, no collision)
          expect(updatedConfig['ccaudit-disabled:ghost-mcp']).toBeTruthy();

          // Assert memory file has frontmatter
          const updatedMemory = await readFile(memoryPath, 'utf8');
          expect(updatedMemory).toMatch(/ccaudit-stale: true/);
          expect(updatedMemory).toMatch(/ccaudit-flagged:/);
        });
      });
    });
