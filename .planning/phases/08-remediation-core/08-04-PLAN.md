    **ManifestWriter class:**
    ```typescript
    /**
     * Append-only JSONL writer with per-op fsync (D-09).
     *
     * Lifecycle:
     *   const w = new ManifestWriter(resolveManifestPath());
     *   await w.open(buildHeader(...));
     *   for (op of ops) await w.writeOp(op);
     *   await w.close(buildFooter(...));  // or close(null) on failure
     */
    export class ManifestWriter {
      private fd: FileHandle | null = null;
      private startMs = 0;

      constructor(public readonly filePath: string) {}

      async open(header: ManifestHeader): Promise<void> {
        // D-10: directory mode 0o700
        await mkdir(path.dirname(this.filePath), { recursive: true, mode: 0o700 });
        // D-10: file mode 0o600 via open() mode arg (applied on create)
        this.fd = await open(this.filePath, 'a', 0o600);
        // Belt-and-suspenders chmod (no-op on Windows, swallow errors)
        try { await chmod(this.filePath, 0o600); } catch { /* swallow */ }
        this.startMs = Date.now();
        await this.fd.write(JSON.stringify(header) + '\n');
        await this.fd.sync();
      }

      async writeOp(op: ManifestOp): Promise<void> {
        if (!this.fd) throw new Error('ManifestWriter.writeOp: not opened (call open() first)');
        // Pitfall 5: concatenate JSON + newline in ONE write to avoid partial-line races
        await this.fd.write(JSON.stringify(op) + '\n');
        await this.fd.sync();
      }

      async close(footer: ManifestFooter | null): Promise<void> {
        if (!this.fd) return;
        if (footer !== null) {
          await this.fd.write(JSON.stringify(footer) + '\n');
          await this.fd.sync();
        }
        await this.fd.close();
        this.fd = null;
      }

      get elapsedMs(): number {
        return Date.now() - this.startMs;
      }
    }
    ```

    **Reader (crash-tolerant):**
    ```typescript
    export interface ReadManifestResult {
      header: ManifestHeader | null;
      ops: ManifestOp[];
      footer: ManifestFooter | null;
      truncated: boolean; // true if trailing line was skipped
    }

    /**
     * Read a manifest JSONL file and parse each line as a discriminated record.
     *
     * Per D-09 crash-survival contract: tolerate a single truncated trailing line
     * (SIGKILL after partial write). Phase 9 uses:
     *   header present + footer present → clean bust, restore normally
     *   header present + footer missing → partial bust, warn and restore what was recorded
     *   header missing → corrupt, refuse
     */
    export async function readManifest(filePath: string): Promise<ReadManifestResult> {
      const raw = await readFile(filePath, 'utf8');
      const lines = raw.split('\n');
      // Strip trailing empty line from final '\n'
      while (lines.length > 0 && lines[lines.length - 1] === '') lines.pop();

      let header: ManifestHeader | null = null;
      const ops: ManifestOp[] = [];
      let footer: ManifestFooter | null = null;
      let truncated = false;

      for (let i = 0; i < lines.length; i++) {
        const line = lines[i]!;
        try {
          const parsed = JSON.parse(line) as ManifestRecord;
          if ('record_type' in parsed && parsed.record_type === 'header') {
            header = parsed as ManifestHeader;
          } else if ('record_type' in parsed && parsed.record_type === 'footer') {
            footer = parsed as ManifestFooter;
          } else if ('op_type' in parsed) {
            ops.push(parsed as ManifestOp);
          }
        } catch {
          // Trailing truncated line is tolerable ONLY if it's the last line
          if (i === lines.length - 1) {
            truncated = true;
          } else {
            // Mid-file parse error is corruption — surface as throw
            throw new Error(`Manifest parse error at line ${i + 1}: invalid JSON`);
          }
        }
      }

      return { header, ops, footer, truncated };
    }
    ```
