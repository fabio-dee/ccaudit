    ```typescript
    import { glob } from 'tinyglobby';
    import path from 'node:path';
    import type { InventoryItem } from './types.ts';

    ```

    Export async function `scanAgents(claudePaths: ClaudePaths, projectPaths: string[]): Promise<InventoryItem[]>`:

    1. Initialize empty `items: InventoryItem[]`
    2. For each base path in `[claudePaths.legacy, claudePaths.xdg]`:
       - Convert to posix path: `const posixBase = base.replace(/\\/g, '/')`
       - Call `glob([posixBase + '/agents/**/*.md'], { absolute: true, dot: false })` wrapped in try/catch (return empty on error)
       - For each discovered file: push `{ name: path.basename(filePath, '.md'), path: filePath, scope: 'global', category: 'agent', projectPath: null }`
    3. For each `projPath` in `projectPaths`:
       - Build `agentsDir = path.join(projPath, '.claude', 'agents')`
       - Convert to posix: `posixDir = agentsDir.replace(/\\/g, '/')`
       - Call `glob([posixDir + '/**/*.md'], { absolute: true, dot: false })` in try/catch
       - For each file: push `{ name: path.basename(filePath, '.md'), path: filePath, scope: 'project', category: 'agent', projectPath: projPath }`
    4. Return `items`

    Export async function `resolveSkillName(skillDir: string): Promise<string>`:
    1. Read `path.join(skillDir, 'SKILL.md')` in try/catch
    2. Match `/^name:\s*(.+)$/m` against content
    3. Return matched group trimmed, or `path.basename(skillDir)` as fallback

    Export async function `scanSkills(claudePaths: ClaudePaths, projectPaths: string[]): Promise<InventoryItem[]>`:
    1. Initialize empty `items: InventoryItem[]`
    2. For each base path in `[claudePaths.legacy, claudePaths.xdg]`:
       - Build `skillsDir = path.join(base, 'skills')`
       - Call `readdir(skillsDir, { withFileTypes: true })` in try/catch (continue on error)
       - For each entry: skip if `entry.name.startsWith('.')`, skip if not `entry.isDirectory()` and not `entry.isSymbolicLink()`
       - Push `{ name: entry.name, path: path.join(skillsDir, entry.name), scope: 'global', category: 'skill', projectPath: null }`
    3. For each `projPath` in `projectPaths`:
       - Build `skillsDir = path.join(projPath, '.claude', 'skills')`
       - Same readdir + filter logic
       - Push with `scope: 'project'`, `projectPath: projPath`
    4. Return `items`

    Define internal `ClaudeConfig` interface:
    ```typescript
    interface ClaudeConfig {
      mcpServers?: Record<string, unknown>;
      projects?: Record<string, {
        mcpServers?: Record<string, unknown>;
        disabledMcpServers?: string[];
      }>;
      skillUsage?: Record<string, {
        usageCount: number;
        lastUsedAt: number;
      }>;
    }
    ```

    Export async function `readClaudeConfig(configPath?: string): Promise<ClaudeConfig>`:
    1. Default `configPath` to `path.join(homedir(), '.claude.json')`
    2. `readFile(configPath, 'utf-8')` in try/catch
    3. `JSON.parse(raw)` in try/catch
    4. Return parsed object or `{}` on any error

    Export async function `scanMcpServers(claudeConfigPath: string | undefined, projectPaths: string[]): Promise<InventoryItem[]>`:
    1. Call `readClaudeConfig(claudeConfigPath)` to get `config`
    2. Initialize empty `items: InventoryItem[]` and `seen = new Set<string>()` for dedup
    3. **Global mcpServers** (root level): iterate `Object.keys(config.mcpServers ?? {})`. For each `serverName`: push `{ name: serverName, path: claudeConfigPath ?? path.join(homedir(), '.claude.json'), scope: 'global', category: 'mcp-server', projectPath: null }`. Add `global::${serverName}` to `seen`.
    4. **Per-project mcpServers**: iterate `Object.entries(config.projects ?? {})`. For each `[projPath, projConfig]`, iterate `Object.keys(projConfig.mcpServers ?? {})`. For each `serverName`: compute dedup key `${projPath}::${serverName}`. If not in `seen`, push `{ name: serverName, path: claudeConfigPath ?? path.join(homedir(), '.claude.json'), scope: 'project', category: 'mcp-server', projectPath: projPath }`. Add to `seen`.
    5. **`.mcp.json` files**: for each `projPath` in `projectPaths`, read `path.join(projPath, '.mcp.json')` in try/catch, parse JSON. For each key in `parsed.mcpServers ?? {}`: compute dedup key `${projPath}::${serverName}`. If not in `seen`, push `{ name: serverName, path: path.join(projPath, '.mcp.json'), scope: 'project', category: 'mcp-server', projectPath: projPath }`. Add to `seen`.
    6. Return `items`

    Note: The `disabledMcpServers` array is read as part of `ClaudeConfig` but not used for filtering here -- it will be used in Plan 03 coordinator for annotation. The `readClaudeConfig` function is exported so Plan 03 can access `skillUsage` and `disabledMcpServers`.
